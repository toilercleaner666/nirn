fighters_guild_faction = {
	monarch_power = MIL
	always = yes

	modifier = {
		discipline = 0.05
		global_regiment_recruit_speed = -0.1
		global_regiment_cost = -0.15
		improve_relation_modifier = -0.1
	}

	triggered_faction_name = {
		name = companions_faction

		trigger = {
			AND = {
				OR = {
					AND = {
						culture_group = nordic
						NOT = { culture = bruman }
						NOT = { culture = horunian }
						capital_scope = {
							superregion = skyrim_superregion
						}
					}
					AND = {
						culture_group = reachmen_group
						capital_scope = {
							superregion = skyrim_superregion
						}
					}
				}
				owns = 86
			}
		}
	}
}

# Knightly Orders of Daggerfall, Cyrodiil, Hammerfell
# Though the modifiers are regional, the faction is global
# You just have to own a province in any of those regions to get the faction

knightly_orders_faction = {
	monarch_power = MIL

	always = yes

	trigger = {
		any_owned_province = {
			OR = {
				superregion = high_rock_superregion
				superregion = cyrodiil_superregion
				superregion = hammerfell_superregion
			}
		}
	}

	modifier = {
		discipline = 0.05
		global_regiment_recruit_speed = -0.1
		global_regiment_cost = -0.15
		improve_relation_modifier = -0.1
	}
}
